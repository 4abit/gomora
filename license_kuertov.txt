using System;
using System.Net.Http;
using System.Text.Json;
using System.Threading.Tasks;

public class LicenseData
{
    public string username { get; set; }
    public string licenseKey { get; set; }
    public string expiry { get; set; }
}

public static class LicenseManager
{
    private const string URL =
        "https://raw.githubusercontent.com/4abit/gomora/main/license.json";

    public static async Task<bool> ValidateLicense(string enteredKey)
    {
        try
        {
            using HttpClient client = new HttpClient();

            string json = await client.GetStringAsync(URL);

            LicenseData data =
                JsonSerializer.Deserialize<LicenseData>(json);

            if (data == null)
                return false;

            if (data.licenseKey != enteredKey)
                return false;

            DateTime expiry =
                DateTime.Parse(data.expiry);

            if (DateTime.Now > expiry)
                return false;

            Console.WriteLine($"Welcome {data.username}");

            return true;
        }
        catch
        {
            return false;
        }
    }
}
